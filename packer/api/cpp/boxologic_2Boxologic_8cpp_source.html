239

241  switch (j)

243  case 1:
244  ex_dim = box.width;
245  dimen2 = box.height;
246  dimen3 = box.length;
247  break;
248  case 2:
249  ex_dim = box.height;

251  dimen3 = box.length;
252  break;
253  case 3:
254  ex_dim = box.length;
255  dimen2 = box.width;
256  dimen3 = box.height;
257  break;

669 void Boxologic::find_layer(double thickness)
670 {
671  // MINIMUM SUM OF LAYERS (height)
672  double min_eval = INT_MAX;
673 
674  layer_thickness = 0;
675 
676  for (size_t i = 0; i < box_array.size(); i++)
677  {
678  const struct Box &box = box_array[i];
679  if (box.is_packed)
680  continue;
681 
682  for (size_t j = 1; j <= 3; j++)
683  {
684  double ex_dim; // STANDARD LENGTH ON THE DIMENSION
685  double dim2; // THE SECOND, LENGTH ON A RESIDUAL DIMENSION
686  double dim3; // THE THIRD, LENGTH ON A RESIDUAL DIMENSION
687 
688  double my_eval = 0;
689 
690  // FETCH STANDARD DIMENSIONS FROM EACH AXIS
691  switch (j)
692  {
693  case 1:
694  ex_dim = box.width;
695  dim2 = box.height;
696  dim3 = box.length;
697  break;
698  case 2:
699  ex_dim = box.height;
700  dim2 = box.width;
701  dim3 = box.length;
702  break;
703  case 3:
704  ex_dim = box.length;
705  dim2 = box.width;
706  dim3 = box.height;
707  break;
708  }
709 
710  // ABOUT ALL BOXES, FIND THE MINIMUM LENGTH OF GAP ~,
711  // STACK ON THE CURRENT LAYER (ADD ON TO THE LAYER_EVAL)
712  if (
713  ex_dim <= thickness &&
714  (
715  (dim2 <= pallet.layout_width && dim3 <= pallet.layout_length) ||
716  (dim3 <= pallet.layout_width && dim2 <= pallet.layout_length)
717  )
718  )
719  {
720  for (size_t k = 0; k < box_array.size(); k++)
721  {
722  const struct Box &my_box = box_array[k];
723 
724  // SAME INSTANCE WITH THE SAME INDEX OR ALREADY PACKED
725  if (i == k || my_box.is_packed == true)
726  continue;
727 
728  double dim_diff =
729  std::min
730  ({
731  abs(ex_dim - my_box.width),
732  abs(ex_dim - my_box.height),
733  abs(ex_dim - my_box.length)
734  });
735  my_eval += dim_diff;
736  }
737  if (my_eval < min_eval)
738  {
739  min_eval = my_eval;
740  layer_thickness = ex_dim;
741  }
742  }

781 void Boxologic::analyze_box(size_t index, double hmx, double hy, double hmy, double hz, double hmz, double dim1, double dim2, double dim3)
782 {
783  // OUT OF BOUNDARY RANGE

785  return;
786

788  (
789  hy - dim2 < bfy ||

814  bboxz = dim3;

816  bbfy = dim2 - hy;

858 void Boxologic::check_found()
859 {
860  evened = false;
861 
862  if (boxi != (size_t)-1)
863  {
864  cboxi = boxi;
865  cbox_layout_width = boxx;
866  cbox_layout_height = boxy;
867  cbox_layout_length = boxz;
868  }
869  else
870  {
871  auto &prev = fetch_scrap_min_z_left();
872  auto &next = fetch_scrap_min_z_right();
873 
874  if (bboxi != (size_t)-1 && // IN RANGE
875  (
876  layer_in_layer != 0 ||
877  (
878  // NO LEFT AND RIGHT EXISTS
879  prev == scrap_list.end() && next == scrap_list.end()
880  )
881  ))
882  {
884  // ~ OR SCRAP_MIN_Z HAS NO NEIGHBOR
886  if (layer_in_layer == 0)
887  {
888  pre_layer = layer_thickness;
889  lilz = scrap_min_z->cumz;
890  }
891 
892  cboxi = bboxi;
893  cbox_layout_width = bboxx;
894  cbox_layout_height = bboxy;
895  cbox_layout_length = bboxz;
896 
897  layer_in_layer += bboxy - layer_thickness;
898  layer_thickness = bboxy;
899  }
900  else
901  {
902  if (prev == scrap_list.end() && next == scrap_list.end())
903  {
905  // SCRAP_MIN_Z HAS NO NEIGHBOR
907  // IN RANGE & NO NEIGHBOR
908  // LAYER HAS DONE.
909  layer_done = true;
910  }
911  else
912  {
913  evened = true;
914 
915  if (prev == scrap_list.end())
916  {
918  // NO LEFT, BUT RIGHT
920  // ERASE SCRAP_MIN_Z
921  // RIGHT IS THE NEW SCRAP_MIN_Z
922  scrap_min_z = scrap_list.erase(scrap_min_z);
923  }
924  else if (next == scrap_list.end())
925  {
927  // NO RIGHT, BUT LEFT
929  // ERASE CURRENT SCRAP_MIN_Z
930  // THE LEFT ITEM FETCHES MIN'S CUM_X
931  prev->cumx = scrap_min_z->cumx;
932 
933  // ERASE FROM MIN_Z TO END
934  scrap_min_z = scrap_list.erase(scrap_min_z, scrap_list.end());
935  }
936  else
937  {
939  // LEFT & RIGHT ARE ALL EXIST
941  if (prev->cumz == next->cumz)
942  {
943  // ----------------------------------------
944  // LEFT AND RIGHT'S CUM_Z ARE EQUAL
945  // ----------------------------------------
946  // LEFT FETCHES THE RIGHT'S CUM_X
947  prev->cumx = next->cumx;
948 
949  // ERASE MIN AND ITS RIGHT
950  auto next_of_next = next;
951  next_of_next++;
952 
953  scrap_min_z = scrap_list.erase(scrap_min_z, next_of_next);
954  }
955  else
956  {
957  // ----------------------------------------
958  // LEFT AND RIGHT'S CUM_Z ARE NOT EQUAL
959  // ----------------------------------------
960  if (prev->cumz == next->cumz)
961  prev->cumx = scrap_min_z->cumx;
962 
963  // ERASE SCRAP_MIN_Z
964  scrap_min_z = scrap_list.erase(scrap_min_z);
965  }
966  }
967  }
968  }
969  }
