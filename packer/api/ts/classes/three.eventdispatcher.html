Class EventDispatcher

JavaScript events for custom objects

Example

var Car = function () {

    EventDispatcher.call( this );
    this.start = function () {

        this.dispatchEvent( { type: 'start', message: 'vroom vroom!' } );

    };

};

var car = new Car();
car.addEventListener( 'start', function ( event ) {

    alert( event.message );

} );
car.start();
